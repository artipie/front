To install or publish NuGet package into Artipie, specify repository url and credentials in NuGet.Config xml file: 
    <?xml version="1.0" encoding="utf-8"?>
        <configuration>
            <packageSources>
                <add key="artipie-repo" value="https://central.artipie.com/{{user}}/{{repo}}/index.json" />
            </packageSources>
            <packageSourceCredentials>
                <artipie-nuget-test>
                    <add key="Username" value="{{user}}"/>
                    <add key="ClearTextPassword" value="****"/>
                </artipie-nuget-test>
            </packageSourceCredentials>
        </configuration>
Then use this configuration when installing or publishing NuGet packages: 
nuget push ./nuget-package -ConfigFile ./NuGet.Config -Verbosity detailed -Source artipie-repo
nuget install Newtonsoft.Json -Version 12.0.3 -NoCache -ConfigFile ./NuGet.Config -Source artipie-repo
